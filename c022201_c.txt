#include<stdio.h>
#include <math.h>
#include "glibw32.h"

main()
{
    ginit(500, 500, WHITE);
    GRAPH g;
    g.window(-3.14, -3.14, 3.14, 3.14);
    g.line(-3.14, 0, 3.14, 0);
    g.line(0, 3.14, 0, -3.14);
    g.setcolor(BLACK);
    for(int i = -180; i < 180; i++)
    {
        double r = (i / 180) * pi;
        double x = r;
        double y = sin(x);
        g.pset(x, y)
    }
}
