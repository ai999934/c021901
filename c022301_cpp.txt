#include <stdio.h>
#include <math.h>
#include "glibw32.h"

main()
{
    int i;
    double x, y, r;
    ginit(500, 500, WHITE);
    GRAPH g;
    g.window(-3.14, -3.14, 3.14, 3.14);
    g.line(-3.14, 0, 3.14, 0);
    g.line(0, 3.14, 0, -3.14);
    g.setcolor(BLACK);
    for(i = -180; i < 180; i++)
    {
        r = (i / 180) * 3.14;
        x = r;
        y = x;
        g.pset(x, y);
    }
    gend();
}
